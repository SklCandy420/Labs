	   DCR B
	   JNZ FACT
	   INX H
	   MOV M,D
	   MVI D,07
	   INX H
	   MOV M,D
	   CALL DIV
	   HLT

DIV:	   LXI H,2001
	   MOV A,M
	   MVI C,00
	   INX H
	   MOV B,M

LOOP:	   CMP B
	   JC CARRY
	   SUB B
	   INR C
	   JMP LOOP

CARRY:	   STA 3050
	   MOV A,C
	   STA 3051
	   HLT

MULT:	   MOV E,B
	   MVI A,00

MULTLOOP:  ADD D
	   DCR E
	   JNZ MULTLOOP
	   MOV D,A
	   RET
